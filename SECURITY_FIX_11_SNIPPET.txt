if (preReg) {
  // SECURITY FIX #11: Ensure conversion flag update succeeds
  const { error: updateError } = await supabase
    .from(preRegTable)
    .update({
      converted: true,
      converted_at: new Date().toISOString(),
    })
    .eq('id', preReg.id);

  if (updateError) {
    console.error(
      `[CompetitionDetail] Failed to mark pre-registration as converted: ${updateError.message}`
    );
    // Don't proceed with payment if we can't mark the pre-reg
    throw new Error(
      'Failed to process pre-registration. Please try again.'
    );
  }
}
